# P14994 异或最短路和

## 题目描述

带权无向图，d_uv=u到v的异或最短路。求所有点对的d_uv之和对998244353取模。

## 输入格式

第一行 n m。
接下来 m 行：u_i v_i w_i。

## 输出格式

一个整数。

## 解题思路

### 核心思路
- 找连通分量
- 每个连通分量内建线性基
- 利用线性基计算点对异或最短路之和

### 复杂度
- 时间: O((n+m) * 30)
- 空间: O(n + m)

## 参考代码

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>
using namespace std;

const int MAXN = 200005;
const long long MOD = 998244353;

int n, m;
vector<pair<int, long long>> adj[MAXN];
int visited[MAXN];
int component[MAXN];
int compCnt;
long long dist[MAXN];

void dfs(int u, int comp) {
    component[u] = comp;
    visited[u] = 1;
    for (auto& [v, w] : adj[u]) {
        if (!visited[v]) { dist[v] = dist[u] ^ w; dfs(v, comp); }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        int u, v; long long w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }
    compCnt = 0;
    memset(visited, 0, sizeof(visited));
    for (int i = 1; i <= n; i++) {
        if (!visited[i]) { compCnt++; dist[i] = 0; dfs(i, compCnt); }
    }
    long long result = 0;
    for (int comp = 1; comp <= compCnt; comp++) {
        vector<int> nodes;
        for (int i = 1; i <= n; i++) if (component[i] == comp) nodes.push_back(i);
        if (nodes.size() <= 1) continue;
        // Build linear basis from cycle space
        vector<long long> basis(30, 0);
        for (int i = 0; i < (int)nodes.size(); i++) {
            for (auto& [v, w] : adj[nodes[i]]) {
                if (component[v] == comp && v > nodes[i]) {
                    long long cycle = dist[nodes[i]] ^ dist[v] ^ w;
                    for (int b = 29; b >= 0; b--) {
                        if ((cycle >> b) & 1) {
                            if (basis[b] == 0) { basis[b] = cycle; break; }
                            cycle ^= basis[b];
                        }
                    }
                }
            }
        }
        // For each pair, compute minimum XOR using basis
        for (int i = 0; i < (int)nodes.size(); i++) {
            for (int j = i + 1; j < (int)nodes.size(); j++) {
                long long xorVal = dist[nodes[i]] ^ dist[nodes[j]];
                for (int b = 29; b >= 0; b--) if ((xorVal >> b) & 1) xorVal ^= basis[b];
                result = (result + xorVal) % MOD;
            }
        }
    }
    cout << result << "\n";
    return 0;
}
```
