# P2630 - 枚举操作序列

## 题目描述

参见洛谷原题。

## 解题思路

### 问题分析

本题属于 **暴力枚举** 类问题。

### 核心思路

枚举所有A/B/C/D操作组合，找最短字典序最小解。

### 时间复杂度

取决于具体算法实现。

## 参考代码

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

vector<vector<int>> rotate90(vector<vector<int>>& m) {
    int n = m.size();
    vector<vector<int>> r(n, vector<int>(n));
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            r[j][n-1-i] = m[i][j];
    return r;
}

vector<vector<int>> flipH(vector<vector<int>>& m) {
    int n = m.size();
    vector<vector<int>> r(n, vector<int>(n));
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            r[i][n-1-j] = m[i][j];
    return r;
}

vector<vector<int>> flipV(vector<vector<int>>& m) {
    int n = m.size();
    vector<vector<int>> r(n, vector<int>(n));
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            r[n-1-i][j] = m[i][j];
    return r;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    vector<vector<int>> init(3, vector<int>(3));
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++) cin >> init[i][j];
    vector<vector<int>> target(3, vector<int>(3));
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++) cin >> target[i][j];
    // Try all sequences of A,B,C,D
    auto cmp = [](vector<vector<int>>& a, vector<vector<int>>& b) {
        for (int i = 0; i < 3; i++)
            for (int j = 0; j < 3; j++)
                if (a[i][j] != b[i][j]) return false;
        return true;
    };
    string best = "";
    for (int mask = 1; mask < 256; mask++) {
        vector<vector<int>> cur = init;
        string ops = "";
        bool ok = true;
        for (int b = 0; b < 8; b++) {
            if (!(mask & (1 << b))) continue;
            int op = b % 4;
            if (op == 0) { cur = rotate90(cur); ops += "A"; }
            else if (op == 1) { cur = rotate90(cur); cur = rotate90(cur); cur = rotate90(cur); ops += "B"; }
            else if (op == 2) { cur = flipH(cur); ops += "C"; }
            else { cur = flipV(cur); ops += "D"; }
            if (ops.size() > 8) { ok = false; break; }
        }
        if (ok && cmp(cur, target)) {
            if (best.empty() || ops.size() < best.size() || (ops.size() == best.size() && ops < best))
                best = ops;
        }
    }
    if (best.empty()) cout << "Poland cannot into space!!!\n";
    else cout << best << "\n";
    return 0;
}
```
